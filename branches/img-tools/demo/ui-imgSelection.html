ui.imgSelection.js

Artworks are from my favorite street artist, Banksy.

Basic selection

Code

$('img#demo-1').imgSelection({
     onSelect: function(e, ui) {
         var coords = $(this.element).imgSelection('serialize', e.target);
         $('#demo-1-value').val(coords);
     }
});
// or (same as above)
$('img#demo-1').imgSelection({
     input: '#demo-1-value'
});
